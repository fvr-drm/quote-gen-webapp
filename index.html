<script src="https://cdnjs.cloudflare.com/ajax/libs/jspdf/2.5.1/jspdf.umd.min.js"></script>

<script>

  window.onload = () => {

    const { jsPDF } = window.jspdf;


    const jsonInput = document.getElementById('jsonArea');

    const validateBtn = document.getElementById('validateBtn');

    const generateBtn = document.getElementById('generateBtn');

    const downloadBtn = document.getElementById('downloadBtn');

    const status = document.getElementById('status');


    let pdfDoc = null;

    let pdfBlobUrl = null;


    function parseJson() {

      try {

        return JSON.parse(jsonInput.value);

      } catch (e) {

        return null;

      }

    }


    validateBtn.onclick = () => {

      const parsed = parseJson();

      if (parsed) {

        status.textContent = 'Valid JSON';

        status.style.color = 'green';

        generateBtn.disabled = false;

      } else {

        status.textContent = 'Invalid JSON. Please fix syntax errors.';

        status.style.color = 'red';

        generateBtn.disabled = true;

        downloadBtn.disabled = true;

      }

    };


    function addTableRow(doc, y, cols, colWidths, fontSize, isHeader = false) {

      let x = 15;

      doc.setFontSize(fontSize);

      cols.forEach((col, i) => {

        if (isHeader) {

          doc.setFont(undefined, 'bold');

          doc.setTextColor(37, 99, 235);

          doc.setFillColor(230, 240, 255);

          doc.rect(x, y - fontSize, colWidths[i], fontSize + 8, 'F'); // filled header cell

        } else {

          doc.setFont(undefined, 'normal');

          doc.setTextColor(0, 0, 0);

        }

        doc.text(String(col), x + 3, y + 6);

        doc.rect(x, y - fontSize, colWidths[i], fontSize + 8);

        x += colWidths[i];

      });

    }


    function generatePDF(json) {

      pdfDoc = new jsPDF({ unit: 'pt', format: 'a4' });

      const pageWidth = pdfDoc.internal.pageSize.getWidth();

      const margin = 15;

      let y = 40;

      const lineHeight = 18;

      const fontSize = 12;


      // Header - Company Info

      if (json.company) {

        pdfDoc.setFontSize(16);

        pdfDoc.setTextColor('#2563eb');

        if (json.company.name) {

          pdfDoc.text(json.company.name, margin, y);

          y += lineHeight;

        }

        pdfDoc.setFontSize(10);

        pdfDoc.setTextColor('#444');

        if (json.company.address) { pdfDoc.text(json.company.address, margin, y); y += lineHeight; }

        if (json.company.contact) { pdfDoc.text(json.company.contact, margin, y); y += lineHeight; }

        y += 10;

      }


      // Title and Quote/Receipt Info (top right)

      pdfDoc.setFontSize(18);

      pdfDoc.setTextColor('#1e40af');

      const docTitle = json.title || 'Sales Document';

      let titleWidth = pdfDoc.getTextWidth(docTitle);

      pdfDoc.text(docTitle, pageWidth - margin - titleWidth, y);

      y += lineHeight + 5;


      pdfDoc.setFontSize(fontSize);

      pdfDoc.setTextColor('#000');

      if (json.document_number) {

        let txt = `Document #: ${json.document_number}`;

        pdfDoc.text(txt, pageWidth - margin - pdfDoc.getTextWidth(txt), y);

        y += lineHeight;

      }

      if (json.date) {

        let txt = `Date: ${json.date}`;

        pdfDoc.text(txt, pageWidth - margin - pdfDoc.getTextWidth(txt), y);

        y += lineHeight;

      }

      if (json.payment_terms) {

        let txt = `Terms: ${json.payment_terms}`;

        pdfDoc.text(txt, pageWidth - margin - pdfDoc.getTextWidth(txt), y);

        y += lineHeight + 10;

      }


      // Customer Info

      if (json.customer) {

        pdfDoc.setTextColor('#2563eb');

        pdfDoc.setFontSize(14);

        pdfDoc.text('Bill To:', margin, y);

        y += lineHeight;

        pdfDoc.setFontSize(fontSize);

        pdfDoc.setTextColor('#000');

        if (json.customer.name) { pdfDoc.text(json.customer.name, margin, y); y += lineHeight; }

        if (json.customer.address) { pdfDoc.text(json.customer.address, margin, y); y += lineHeight; }

        if (json.customer.contact) { pdfDoc.text(json.customer.contact, margin, y); y += lineHeight + 10; }

      }


      // Items Table Header

      const colWidths = [260, 60, 90, 90];

      let tableY = y;

      addTableRow(pdfDoc, tableY, ['Description', 'Qty', 'Unit Price', 'Amount'], colWidths, fontSize, true);

      y += lineHeight + 8;


      // Items Table Rows

      if (Array.isArray(json.items)) {

        json.items.forEach(item => {

          addTableRow(

            pdfDoc, y,

            [

              item.description || '',

              String(item.quantity || ''),

              item.unit_price ? `$${item.unit_price.toFixed(2)}` : '',

              item.amount ? `$${item.amount.toFixed(2)}` : ''

            ],

            colWidths,

            fontSize

          );

          y += lineHeight + 8;

          // Page break check

          if (y > 760) {

            pdfDoc.addPage();

            y = 40;

            // Repeat table header on new page

            addTableRow(pdfDoc, y, ['Description', 'Qty', 'Unit Price', 'Amount'], colWidths, fontSize, true);

            y += lineHeight + 8;

          }

        });

      }


      // Totals

      y += 15;

      if (json.totals) {

        Object.entries(json.totals).forEach(([key, value]) => {

          const label = key.charAt(0).toUpperCase() + key.slice(1).replace(/_/g, ' ');

          const formattedValue = typeof value === 'number' ? `$${value.toFixed(2)}` : String(value);

          pdfDoc.text(`${label}:`, pageWidth - margin - pdfDoc.getTextWidth(`${label}:`) - 100, y);

          pdfDoc.text(formattedValue, pageWidth - margin - pdfDoc.getTextWidth(formattedValue), y);

          y += lineHeight;

        });

      }


      // Footer

      const footerY = 820;

      pdfDoc.setLineWidth(0.5);

      pdfDoc.setDrawColor(128);

      pdfDoc.line(margin, footerY, pageWidth - margin, footerY);

      pdfDoc.setFontSize(10);

      pdfDoc.setTextColor('#666');

      pdfDoc.text(json.footer_note || 'Thank you for your business!', margin, footerY + 15);

      if (json.company && json.company.website) {

        pdfDoc.text(`Website: ${json.company.website}`, margin, footerY + 30);

      }


      return pdfDoc;

    }


    generateBtn.onclick = () => {

      const parsed = parseJson();

      if (!parsed) {

        status.textContent = 'Invalid JSON. Please fix syntax errors before generating PDF.';

        status.style.color = 'red';

        return;

      }

      status.textContent = 'Generating PDF...';

      status.style.color = 'black';


      pdfDoc = generatePDF(parsed);

      status.textContent = 'PDF Generated. Ready to download.';

      status.style.color = 'green';

      downloadBtn.disabled = false;


      const pdfBlob = pdfDoc.output('blob');

      if (pdfBlobUrl) {

        URL.revokeObjectURL(pdfBlobUrl);

      }

      pdfBlobUrl = URL.createObjectURL(pdfBlob);

      downloadBtn.href = pdfBlobUrl;

      downloadBtn.download = `sales-document-${new Date().toISOString().slice(0,10)}.pdf`;

    };


    downloadBtn.onclick = () => {

      setTimeout(() => {

        if (pdfBlobUrl) {

          URL.revokeObjectURL(pdfBlobUrl);

          pdfBlobUrl = null;

        }

      }, 1000);

    };

  };

</script>
